OOPClass01. Описать класс Abit с данными об абитуриенте: ФИО и три оценки. Предусмотреть: 1) возможность расчета и выдачи средней оценки и 2) возможность вывода одной строкой ФИО абитуриента, трех оценок и средней оценки. Программа должна иллюстрировать работу с этим классом.

Пример решения:

// OOPClass01

type

  Abit = class

    fio: string;

    n1, n2, n3: integer;

   

    constructor(pfio: string; pn1, pn2, pn3: integer);

    begin

      (fio, n1, n2, n3) := (pfio, pn1, pn2, pn3);     

    end;

   

    function Avg: real;

    begin

      result := (n1 + n2 + n3) / 3;

    end;

   

    function Info: string;

    begin

      result := fio + ' ' + n1 + ' ' + n2 + ' ' + n3 + ' ' + Avg;

    end;

  end;

 

BEGIN

  var a := new Abit('Иванов А.Б.', 3, 4, 5);

  PrintLn(a.Avg);

  PrintLn(a.Info);

END.

OOPClass14. В копию программы OOPClass13 внесите изменения: 1) в класс Avto добавьте свойство для чтения и изменения пробега; 2) суммарный пробег вычислять через свойство пробега; 3) заполнять массив в цикле значениями с клавиатуры; 4) вычислять суммарный пробега тоже в цикле.

Пример решения:

// OOPClass14

type

  Avto = class

    num: string;

    probeg: real;

   

    constructor(pnum: string; pprobeg: real);

    begin

      (num, probeg) := (pnum, pprobeg);     

    end;

   

    function Info := num + ' ' + probeg;

   

    function GetNum := num;

    procedure SetNum(pnum: string) := num := pnum;

    property NumAvto: string read GetNum write SetNum;

   

    function GetProbeg := probeg;

    procedure SetProbeg(pprobeg: real) := probeg := pprobeg;

    property Mileage: real read GetProbeg write SetProbeg;

  end;

 

BEGIN

  // Создание массива:

  var k := ReadlnInteger('Количество авто');

  var a := new Avto[k];

 

  // Заполнение массива:

  for var i := 0 to k - 1 do

    a[i] := new Avto(ReadlnString('Номер авто:'), ReadlnReal('Пробег:'));

 

  // Вывод массива:

  for var i := 0 to k - 1 do

    PrintLn(a[i].Info);

 

  // Суммарный пробег для авто с номером num: 

  var num := ReadlnString('Номер искомого авто:');

  var sum := 0.0;

    // Вариант 1:

  for var i := 0 to k - 1 do

    if a[i].NumAvto = num then 

      sum += a[i].Mileage;

  PrintLn(sum);

    // Вариант 2:

  sum := ArrGen(k, i -> a[i].NumAvto = num ? a[i].Mileage : 0).Sum;

  PrintLn(sum);

END.
