ЛР17-3. ООП: Базы данных

 

База данных здесь – это текстовый файл в совокупности с подпрограммами, умеющими:
1) создавать массив из строк, введенных с клавиатуры;
2) записывать/дозаписывать этот массив в файл;
3) вносить изменения в файл: добавлять, удалять, изменять строки;
4) находить и отображать нужную информацию.
Каждая строка файла формируется на основе определенного класса.

OOPdb01. Создать базу данных «Абитуриенты» (фамилия, оценка1, оценка2, оценка3). Обеспечить ввод и вывод данных, удаление, изменение, поиск, вычисление средней оценки. Выдать список абитуриентов со средней оценкой выше заданной.

Пример решения:

// OOPdb01

type

  Abit = class

    fam: string;

    n1, n2, n3: integer;

   

    constructor(pfam: string; pn1, pn2, pn3: integer);

    begin

      (fam, n1, n2, n3) := (pfam, pn1, pn2, pn3);     

    end;

   

    function Avg: real;

    begin

      Result := Round((n1 + n2 + n3) / 3, 2);

    end;

   

    function Info: string;

    begin

      Result := fam + ' ' + n1 + ' ' + n2 + ' ' + n3;

    end;

  end;

 

var

  NameFile := 'Abit.txt';

 

function CreateArrayAbit: array of Abit;

begin                          // Ввод записей с клавиатуры в массив

  var a := ReadSeqStringWhile('Фамилия и три оценки:', x -> x <> '').ToArray;

  result := ArrGen(a.Count, i ->

            new Abit(a[i].ToWords[0],

                     a[i].ToWords[1].ToInteger,

                     a[i].ToWords[2].ToInteger,

                     a[i].ToWords[3].ToInteger));

end;

 

procedure ArrayAbitToFile(a: array of Abit; Append: Boolean := True);

begin                          // Запись/дозапись массива в файл

  var b := a.ConvertAll(x -> x.Info);

  if FileExists(NameFile) and Append then

  begin

    var c := ReadAllLines(NameFile);

    b := c + b;

  end;

  WriteAllLines(NameFile, b);

end;

 

function FindAbit(fam: string): Abit;

begin                          // Поиск по фам

  var a := ReadAllLines(NameFile);

  var b := a.Find(x -> x.ToWords[0] = fam);

  if b = nil then

    Result := nil

  else

  begin

    var c := b.ToWords;

    Result := new Abit(c[0], c[1].ToInteger, c[2].ToInteger, c[3].ToInteger);

  end;

end;

 

function UpdateAbit(fam, newstr: string): boolean;

begin                          // Поиск по фам и Замена новой строкой

  var a := ReadAllLines(NameFile);

  var i := a.FindIndex(x -> x.ToWords[0] = fam);

  if i = -1 then

    Result := false

  else

  begin

    var b := a[:i] + |newstr| + a?[i + 1:];

    WriteAllLines(NameFile, b);

    Result := true

  end;

end;

 

function DeleteAbit(fam: string): boolean;

begin                          // Поиск по фам и Удаление

  var a := ReadAllLines(NameFile);

  var i := a.FindIndex(x -> x.ToWords[0] = fam);

  if i = -1 then

    Result := false

  else

  begin

    var b := a[:i] + a?[i + 1:];

    WriteAllLines(NameFile, b);

    Result := true

  end;

end;

 

procedure ListAbit(v: real :=0);

begin                          // Список со сред.оц. >= v

  var a := ReadAllLines(NameFile);

  var b: array of Abit := ArrGen(a.Count, i ->

            new Abit(a[i].ToWords[0],

                     a[i].ToWords[1].ToInteger,

                     a[i].ToWords[2].ToInteger,

                     a[i].ToWords[3].ToInteger));

  for var i := 0 to b.High do

    if b[i].Avg >= v then

      PrintLn(b[i].Info, b[i].Avg);

end;

 

BEGIN

                             // Ввод записей с клавиатуры в массив:

  var a := CreateArrayAbit;

 

                             // Запись массива в файл:

  ArrayAbitToFile(a);

 

                             // Поиск по фамилии:

  var f := FindAbit('Сидоров');

  if f = nil then

    PrintLn('Не найден')

  else

    PrintLn(f.Info, f.Avg);

 

                             // Поиск по фамилии и Удаление:

  if DeleteAbit('Зайцев') then

    PrintLn('Удаление выполнено')

  else

    PrintLn('Удаление не выполнено');

  

                             // Поиск по фамилии и Замена:

  if UpdateAbit('Петров', 'Петрова 5 5 5') then

    PrintLn('Замена выполнена')

  else

    PrintLn('Замена не выполнена');

  

                             // Список со сред.оц. >= 4.5

  ListAbit(4.5);

  

  Exec('notepad.exe', NameFile);

 

END.
