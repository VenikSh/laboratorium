ZMinmax01. Дано целое число N и набор из N чисел. Найти минимальный и максимальный из элементов данного набора и вывести их в указанном порядке. При N=4    4.75 -3.96 1.89 -1.31   результат: -3.96 4.75.

Примеры решения:

С использованием цикла

С использованием последовательностей

##

// ZMinmax01

var n := ReadLnInteger('n=');

var min := real.MaxValue;

var max := real.MinValue;

 

loop n do

begin

  var a := ReadReal;

  if a < min then

    min := a;

  if a > max then

    max := a; 

end;

 

Print(min, max);

##

// ZMinmax01_2

var n := ReadLnInteger('n=');

var min := real.MaxValue;

var max := real.MinValue;

 

var q := ReadSeqReal(n);

 

q.ForEach(x -> begin

               if x < min then

                 min := x;

               if x > max then

                 max := x;

               end);

 

Print(min, max);

 

##

var n := ReadLnInteger('n=');

var min := real.MaxValue;

var max := real.MinValue;

 

var q := ReadSeqReal(n);

 

foreach var x in q do

begin

  if x < min then

    min := x;

  if x > max then

    max := x;

end;

 

Print(min, max);

 

##

var q := Seq(4.75, -3.96, 1.89, -1.31);

Print(q.Min, q.Max);

ZMinmax06. Дано целое число N и набор из N целых чисел. Найти индексы первого минимального и последнего максимального элемента из данного набора и вывести их в указанном порядке. Индексация – с нуля! При N=6    1   0   3   0   2   3   результат:    1   5.
a := ReadArrInteger(n);   a.IndexMin;   a.LastIndexMax

Пример решения:

С использованием цикла

С использованием массива

##

// ZMinmax06

var n := ReadLnInteger('n=');

var (min, max) := (real.MaxValue, real.MinValue);

var (imin, imax) := (0, 0);

for var i := 0 to n - 1 do

begin

  var a := ReadInteger;

  if a < min then

  begin

    min := a;

    imin := i;

  end;

  if a >= max then

  begin

    max := a; 

    imax := i;

  end;

end;

Print(imin, imax);

##

//ZMinmax06_2

var n := ReadlnInteger('n=');

 

var a := ReadArrInteger(n);

 

a.IndexMin.Print;

a.LastIndexMax.Print;
